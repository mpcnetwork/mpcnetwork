Intel “Alchemist” Grafische Kaarten
De Intel Arc Alchemist grafische kaarten zijn de eerste echte dedicated GPU’s van Intel sinds de i740 in 1998. Ze zijn gebaseerd op de Xe HPG microarchitectuur en zijn ontworpen om te concurreren met de GPU’s van AMD en Nvidia. De Arc Alchemist serie omvat verschillende modellen zoals de Arc A770, A750, A580, en A38012. Deze kaarten bieden prestaties die vergelijkbaar zijn met de Nvidia RTX 3060, maar tegen een lagere prijs en met meer VRAM1.

Intel “Battlemage” Grafische Kaarten
De volgende generatie na Alchemist is de “Battlemage” serie. Hoewel er nog geen officiële releasedatum is, wordt verwacht dat deze GPU’s aanzienlijke verbeteringen zullen brengen in vergelijking met de Alchemist serie. De Battlemage GPU’s zullen waarschijnlijk gebruik maken van een nieuwe architectuur met meer Xe cores en een grotere L2 cache34. Er zijn geruchten dat de topmodellen van Battlemage tot 32 Xe2 cores zullen hebben, wat neerkomt op 4096 stream processors3.

import cv2
from flask import Flask, render_template, Response

app = Flask(__name__)

def gen_frames():
    camera = cv2.VideoCapture(0)  # Verander 0 naar de URL van je IP-camera
    while True:
        success, frame = camera.read()
        if not success:
            break
        else:
            ret, buffer = cv2.imencode('.jpg', frame)
            frame = buffer.tobytes()
            yield (b'--frame\r\n'
                   b'Content-Type: image/jpeg\r\n\r\n' + frame + b'\r\n')

@app.route('/video_feed')
def video_feed():
    return Response(gen_frames(), mimetype='multipart/x-mixed-replace; boundary=frame')

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=5000)

import cv2

def detect_motion(frame, avg):
    gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
    gray = cv2.GaussianBlur(gray, (21, 21), 0)
    if avg is None:
        avg = gray.copy().astype("float")
        return avg, False
    cv2.accumulateWeighted(gray, avg, 0.5)
    frameDelta = cv2.absdiff(gray, cv2.convertScaleAbs(avg))
    thresh = cv2.threshold(frameDelta, 25, 255, cv2.THRESH_BINARY)[1]
    thresh = cv2.dilate(thresh, None, iterations=2)
    cnts, _ = cv2.findContours(thresh.copy(), cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
    motion_detected = len(cnts) > 0
    return avg, motion_detected

avg = None
camera = cv2.VideoCapture(0)
while True:
    success, frame = camera.read()
    if not success:
        break
    avg, motion_detected = detect_motion(frame, avg)
    if motion_detected:
        cv2.imwrite('motion_frame.jpg', frame)

import pythreejs as p3
from IPython.display import display

# Maak een eenvoudige 3D kubus
geometry = p3.BoxGeometry(width=1, height=1, depth=1)
material = p3.MeshBasicMaterial(color='red')
cube = p3.Mesh(geometry=geometry, material=material)

# Maak een scène en voeg de kubus toe
scene = p3.Scene(children=[cube, p3.AmbientLight(color='#777777')])
camera = p3.PerspectiveCamera(position=[5, 3, 5], fov=20)
renderer = p3.Renderer(camera=camera, scene=scene, controls=[p3.OrbitControls(controlling=camera)])

display(renderer)
